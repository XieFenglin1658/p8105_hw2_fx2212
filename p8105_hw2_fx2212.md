### comparing rental prices in January 2021 to prices in January 2020

``` r
rent_jan_2020 = rent_data |> 
  filter(year(date) == 2020, month(date) == 1)

rent_jan_2021 = rent_data  |> 
  filter(year(date) == 2021, month(date) == 1)

rent_change = rent_jan_2020  |> 
  select(zip_code, borough, neighborhood, rent_2020 = rent_price)  |> 
  inner_join(
    rent_jan_2021  |>  
      select(zip_code, rent_2021 = rent_price),
    by = "zip_code"
  )  |> 
  mutate(
    rent_change = rent_2021 - rent_2020,
    percent_change = (rent_2021 - rent_2020) / rent_2020 * 100
  )

top_drops = rent_change  |> 
  arrange(percent_change)  |> 
  head(10)

top_drops_table = top_drops  |> 
  select(
    ZIP_Code = zip_code,
    Borough = borough,
    Neighborhood = neighborhood,
    `Rent_Jan_2020` = rent_2020,
    `Rent_Jan_2021` = rent_2021,
    `Price_Drop` = rent_change,
    `Percent_Drop` = percent_change
  ) |> 
  mutate(
    across(c(`Rent_Jan_2020`, `Rent_Jan_2021`, `Price_Drop`), ~round(., 2)),
    `Percent_Drop` = round(`Percent_Drop`, 2)
  )

kable(top_drops_table, 
      caption = "Top 10 ZIP Codes with Largest Rent Drops from Jan 2020 to Jan 2021")
```
